    <script>
        // Функционал поиска и фильтрации
        document.addEventListener('DOMContentLoaded', function() {
            const searchInput = document.getElementById('searchInput');
            const filterButtons = document.querySelectorAll('.filter-btn');
            const subjectCards = document.querySelectorAll('.subject-card');
            
            // Поиск
            searchInput.addEventListener('input', function() {
                const searchTerm = this.value.toLowerCase();
                
                subjectCards.forEach(card => {
                    const title = card.querySelector('.subject-title').textContent.toLowerCase();
                    const materials = card.textContent.toLowerCase();
                    
                    if (title.includes(searchTerm) || materials.includes(searchTerm)) {
                        card.style.display = 'block';
                    } else {
                        card.style.display = 'none';
                    }
                });
            });
            
            // Фильтрация
            filterButtons.forEach(button => {
                button.addEventListener('click', function() {
                    const filter = this.getAttribute('data-filter');
                    
                    // Активный класс
                    filterButtons.forEach(btn => btn.classList.remove('active'));
                    this.classList.add('active');
                    
                    subjectCards.forEach(card => {
                        if (filter === 'all') {
                            card.style.display = 'block';
                        } else {
                            const hasMaterials = card.querySelector(`.material-links a, .material-links .no-materials`);
                            const materialType = card.querySelector(`.material-type:nth-child(${filter === 'lecture' ? 2 : filter === 'practice' ? 3 : 4})`);
                            
                            if (materialType && materialType.querySelector('.material-links a')) {
                                card.style.display = 'block';
                            } else {
                                card.style.display = 'none';
                            }
                        }
                    });
                });
            });
        });
    </script>
