        function updateCountdowns() {
            const now = new Date();
            
            deadlines.forEach(deadline => {
                const targetDate = new Date(deadline.date);
                const timeDiff = targetDate - now;
                
                if (timeDiff <= 0) {
                    document.getElementById(deadline.id).textContent = "–ó–∞–≤–µ—Ä—à—ë–Ω";
                    document.getElementById(deadline.id).className = "countdown completed";

                const hours = Math.floor((timeDiff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((timeDiff % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((timeDiff % (1000 * 60)) / 1000);
                
                const countdownText = `${days} –¥. ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
                
                const countdownElement = document.getElementById(deadline.id);

                if (days < 2) {
                    countdownElement.className = "countdown urgent";
                } else if (days < 7) {
                    countdownElement.className = "countdown warning";
                } else {
                    countdownElement.className = "countdown normal";
                }
            });
